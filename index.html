Las OS se generan mediante solicitud del área correspondiente. La codificación es OS-XX-YYYY donde XX indica el área destino (MC: Taller Mecánico · MT: Taller Metalúrgico · CA: Campamento) e YYYY es la numeración correlativa anual.

📋 Apps Script — Copiar en Google Sheets

1. En Google Drive creá una nueva Hoja de Cálculo.
2. Menú Extensiones → Apps Script → pegá el código de abajo.
3. Implementar → Nueva implementación → Aplicación web → "Cualquier persona".
4. Copiá la URL y pegala arriba.

// ══════════════════════════════════════════════════════════════════ // NIELSEN LOGÍSTICA — Google Apps Script v4.0 // Hojas: OS · VEHICULOS · FOTOS · USUARIOS // Implementar: Ejecutar como YO · Acceso CUALQUIER PERSONA // ══════════════════════════════════════════════════════════════════ const SHEET_OS = "OS"; const SHEET_VEH = "VEHICULOS"; const SHEET_FOTOS = "FOTOS"; const SHEET_USUARIOS= "USUARIOS"; const DRIVE_FOLDER = "Nielsen MTTO Fotos"; // ── CABECERAS POR HOJA ──────────────────────────────────────────── const HEADERS = { OS: ['id','numero','fecha_pedido','tipo','vehiculo_id','vehiculo_label', 'solicitante','area_solicitante','responsable','area_destino', 'plazo','plazo_horas','plazo_unidad','fecha_vencimiento', 'prioridad','timestamp_inicio','timestamp_fin', 'fecha_inicio','fecha_fin','duracion_hs','proximo_servicio', 'km_actuales','km_cierre','desc_trabajo','trabajo_realizado', 'observaciones','materiales','mano_obra','estado','obs_final','tecnico', 'checklist','checklist_nivel'], VEHICULOS: ['id','tipo','marca','modelo','interno','dominio','anio','color', 'km_actuales','km_ultimo_service','km_intervalo', 'fecha_ultimo_service','fecha_proximo_service', 'estado_manual','activo','notas','photo_url','plan_preventivo'], USUARIOS: ['ID','NOMBRE','USUARIO','ROL','EMAIL','PASSWORD_HASH','ACTIVO','CREADO'], FOTOS: ['ID','OS_ID','FILE_ID','FILENAME','CAPTION','THUMB_URL','VIEW_URL','SIZE_KB','TIMESTAMP'] }; // ── OBTENER HOJA CON CABECERAS GARANTIZADAS ─────────────────────── function getSheet(ss, name) { let sh = ss.getSheetByName(name); if (!sh) sh = ss.insertSheet(name); const expected = HEADERS[name]; if (!expected) return sh; if (sh.getLastRow() === 0 || sh.getRange(1,1).getValue() === '') { // Hoja vacía — crear cabeceras completas sh.appendRow(expected); sh.setFrozenRows(1); sh.getRange(1,1,1,expected.length).setBackground('#1a1a2e').setFontColor('#fff').setFontWeight('bold'); if (name === 'USUARIOS') sh.hideColumns(6); } else { // Hoja existente — agregar columnas nuevas que falten (migración automática) const lastCol = sh.getLastColumn(); const current = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(String); const missing = expected.filter(h => !current.includes(h)); if (missing.length) { const startCol = lastCol + 1; sh.getRange(1, startCol, 1, missing.length).setValues([missing]); sh.getRange(1, startCol, 1, missing.length) .setBackground('#1a1a2e').setFontColor('#fff').setFontWeight('bold'); } } return sh; } // ── doGet ───────────────────────────────────────────────────────── function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = e.parameter.sheet || SHEET_OS; const osId = e.parameter.osId; if (sheet === "FOTOS") return jsonOut(getFotos(ss, osId)); if (sheet === "USUARIOS") return jsonOut(getUsuarios(ss)); const sh = getSheet(ss, sheet); const data = sh.getDataRange().getValues(); if (data.length < 2) return jsonOut([]); const headers = data[0]; return jsonOut(data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { let val = row[i]; // Convertir objetos Date a string yyyy-mm-dd para que el HTML los lea bien if (val instanceof Date && !isNaN(val)) { const yyyy = val.getFullYear(); const mm = String(val.getMonth() + 1).padStart(2, '0'); const dd = String(val.getDate()).padStart(2, '0'); val = yyyy + '-' + mm + '-' + dd; } obj[h] = val; }); return obj; })); } // ── doPost ──────────────────────────────────────────────────────── function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'uploadPhoto') return uploadPhoto(data, ss); if (data.action === 'deletePhoto') return deletePhoto(data, ss); if (data.action === 'deletePhotosByOS') return deletePhotosByOS(data, ss); if (data.action === 'updatePhotoCaption') return updateCaption(data, ss); // OS / VEHICULOS / USUARIOS const sheetMap = { VEHICULOS: SHEET_VEH, USUARIOS: SHEET_USUARIOS }; const shName = sheetMap[data.sheet] || SHEET_OS; const sh = getSheet(ss, shName); const rows = sh.getDataRange().getValues(); const headers = rows.length ? rows[0] : []; if (!headers.length) return jsonOut({ ok: false, error: 'Sin cabeceras en hoja ' + shName }); const act = data.action; if (act === 'create') { sh.appendRow(headers.map(h => data.record[h] !== undefined ? data.record[h] : '')); } else if (act === 'update' || act === 'upsert') { const recId = String(data.record?.id || data.record?.ID || data.id || ''); const idx = rows.findIndex((r, i) => i > 0 && String(r[0]) === recId); if (idx > 0) { sh.getRange(idx+1, 1, 1, headers.length) .setValues([headers.map(h => data.record[h] !== undefined ? data.record[h] : '')]); } else if (act === 'upsert') { sh.appendRow(headers.map(h => data.record[h] !== undefined ? data.record[h] : '')); } } else if (act === 'delete') { const delId = String(data.id || data.record?.id || ''); const idx = rows.findIndex((r, i) => i > 0 && String(r[0]) === delId); if (idx > 0) sh.deleteRow(idx + 1); } return jsonOut({ ok: true }); } catch(err) { return jsonOut({ ok: false, error: err.toString() }); } } // ── USUARIOS ────────────────────────────────────────────────────── function getUsuarios(ss) { const sh = ss.getSheetByName(SHEET_USUARIOS); if (!sh || sh.getLastRow() < 2) return []; const rows = sh.getDataRange().getValues(); const headers = rows[0]; // Devolver todos (el cliente filtra activos para login) return rows.slice(1) .filter(r => r[0]) // ignorar filas vacías .map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function crearHojaUsuarios(ss) { return getSheet(ss, SHEET_USUARIOS); } // ── FOTOS ───────────────────────────────────────────────────────── function getFotos(ss, osId) { const sh = ss.getSheetByName(SHEET_FOTOS); if (!sh || sh.getLastRow() < 2) return []; const rows = sh.getDataRange().getValues(); const hdr = rows[0].map(h => h.toString().toLowerCase().replace(/ /g,'_')); return rows.slice(1) .filter(r => r[0] && (!osId || r[1] === osId)) .map(r => { const o = {}; hdr.forEach((h, i) => o[h] = r[i]); return o; }); } function uploadPhoto(data, ss) { const { osId, base64, filename, caption, mimeType, sizeKB, timestamp } = data; const root = getDriveFolder(DRIVE_FOLDER); const osFolders = root.getFoldersByName(osId); const osDir = osFolders.hasNext() ? osFolders.next() : root.createFolder(osId); const blob = Utilities.newBlob(Utilities.base64Decode(base64), mimeType||'image/jpeg', filename||'foto.jpg'); const file = osDir.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const fid = file.getId(); const photoId = Utilities.getUuid(); const thumbUrl = 'https://drive.google.com/thumbnail?id=' + fid + '&sz=w400'; const viewUrl = 'https://drive.google.com/file/d/' + fid + '/view'; const sh = ss.getSheetByName(SHEET_FOTOS) || crearHojaFotos(ss); sh.appendRow([photoId, osId, fid, filename||'foto.jpg', caption||'', thumbUrl, viewUrl, sizeKB||0, timestamp||new Date().toISOString()]); return jsonOut({ ok:true, photo:{ id:photoId, osId, fileId:fid, filename, caption:caption||'', thumbUrl, viewUrl, sizeKB:sizeKB||0, timestamp }}); } function deletePhoto(data, ss) { try { DriveApp.getFileById(data.fileId).setTrashed(true); } catch(e) {} const sh = ss.getSheetByName(SHEET_FOTOS); if (sh) { const rows = sh.getDataRange().getValues(); for (let i = rows.length-1; i >= 1; i--) { if (rows[i][2] === data.fileId) { sh.deleteRow(i+1); break; } } } return jsonOut({ ok: true }); } function deletePhotosByOS(data, ss) { const sh = ss.getSheetByName(SHEET_FOTOS); if (sh) { const rows = sh.getDataRange().getValues(); for (let i = rows.length-1; i >= 1; i--) { if (rows[i][1] === data.osId) { try { DriveApp.getFileById(rows[i][2]).setTrashed(true); } catch(e) {} sh.deleteRow(i+1); } } } try { const r = getDriveFolder(DRIVE_FOLDER); const f = r.getFoldersByName(data.osId); if (f.hasNext()) f.next().setTrashed(true); } catch(e) {} return jsonOut({ ok: true }); } function updateCaption(data, ss) { const sh = ss.getSheetByName(SHEET_FOTOS); if (!sh) return jsonOut({ ok: false }); const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][2] === data.fileId) { sh.getRange(i+1, 5).setValue(data.caption); break; } } return jsonOut({ ok: true }); } function crearHojaFotos(ss) { return getSheet(ss, SHEET_FOTOS); } function getDriveFolder(name) { const f = DriveApp.getFoldersByName(name); return f.hasNext() ? f.next() : DriveApp.createFolder(name); } // crearHojaUsuarios y crearHojaFotos se mantienen por compatibilidad // La lógica de creación con cabeceras ahora está centralizada en getSheet() function jsonOut(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
